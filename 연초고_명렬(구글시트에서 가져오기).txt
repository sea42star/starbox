<div id="loading">데이터를 불러오는 중...</div>

<header>
    <div class="tab-wrap">
        <div id="t1" class="tab-btn active" onclick="setTab('class')">학급 명렬</div>
        <div id="t2" class="tab-btn" onclick="setTab('staff')">교직원 명렬</div>

    <button class="f-btn btn-reset" onclick="resetCheck()">초기화</button>
    <button class="f-btn btn-save" onclick="saveCheck()">저장하기</button>
</footer>

<script>
    // ⚠️ 아래 시트 ID를 본인의 구글 시트 ID로 꼭 변경하세요!
    const SHEET_ID = '1dfpz523jAuIUvPUSTszBXdqwDblFSq2H3LEH47QSuU0'; 
    
    let DATA_STORAGE = { class: [], staff: [] };
    let nowMode = 'class';

    // 구글 시트에서 데이터를 가져오는 함수
    async function fetchSheetData() {
        try {
            // '학급' 시트(gid=0)와 '교직원' 시트(gid=시트에서확인가능)를 가져옵니다.
            // 여기서는 단순화를 위해 첫 번째 시트와 두 번째 시트를 순서대로 가져오는 API 예시를 사용합니다.
            const base = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:csv`;
            
            // 학급 데이터 (첫 번째 탭)
            const resClass = await fetch(`${base}&sheet=학급`);
            const textClass = await resClass.text();
            DATA_STORAGE.class = parseCSV(textClass);

            // 교직원 데이터 (두 번째 탭)
            const resStaff = await fetch(`${base}&sheet=교직원`);
            const textStaff = await resStaff.text();
            DATA_STORAGE.staff = parseCSV(textStaff);

            document.getElementById('loading').style.display = 'none';
            render();
        } catch (e) {
            alert('데이터를 불러오지 못했습니다. 시트 ID와 공유 설정을 확인하세요.');
        }
    }

    // CSV 파싱 (구글 시트 응답 처리)
    function parseCSV(csv) {
        return csv.split('\n').map(row => {
            return row.split(',').map(cell => cell.replace(/"/g, ''));
        }).filter(row => row[0]); // 빈 줄 제거
    }

    function setTab(m) {
        nowMode = m;
        document.getElementById('t1').className = (m === 'class' ? 'tab-btn active' : 'tab-btn');
        document.getElementById('t2').className = (m === 'staff' ? 'tab-btn active' : 'tab-btn');
        document.getElementById('targetTable').className = (m === 'class' ? 'mode-class' : 'mode-staff');
        render();
    }

    function render() {
        const table = document.getElementById('targetTable');
        table.innerHTML = "";
        const data = DATA_STORAGE[nowMode];
        const rowCount = (nowMode === 'class' ? 10 : 20);
        const colCount = (nowMode === 'class' ? 3 : 4);

        for (let r = 0; r < rowCount; r++) {
            const tr = document.createElement('tr');
            const tdNum = document.createElement('td');
            tdNum.className = 'no-cell';
            tdNum.innerText = r + 1;
            tr.appendChild(tdNum);

            for (let c = 0; c < colCount; c++) {
                const tdName = document.createElement('td');
                const idx = r + (c * rowCount);
                if (idx < data.length) {
                    const cell = document.createElement('div');
                    cell.className = 'name-cell';
                    cell.id = `c-${nowMode}-${idx}`;
                    cell.innerText = data[idx][0];
                    cell.dataset.state = 0;
                    cell.onclick = function() {
                        let s = (parseInt(this.dataset.state) + 1) % 3;
                        applyState(this, s);
                    };
                    let timer;
                    cell.ontouchstart = () => {
                        timer = setTimeout(() => {
                            if(confirm(`${data[idx][0]}님께 전화하시겠습니까?\n(${data[idx][1]})`)) {
                                location.href = 'tel:' + data[idx][1];
                            }
                        }, 500);
                    };
                    cell.ontouchend = () => clearTimeout(timer);
                    tdName.appendChild(cell);
                }
                tr.appendChild(tdName);
            }
            table.appendChild(tr);
        }
        load();
    }

    // ... (applyState, saveCheck, load, resetCheck 함수는 이전과 동일)
    function applyState(el, s) {
        el.dataset.state = s;
        el.classList.remove('step-1', 'step-2');
        if (s == 1) el.classList.add('step-1');
        if (s == 2) el.classList.add('step-2');
    }

    function saveCheck() {
        const states = JSON.parse(localStorage.getItem('att_remote_v1') || '{}');
        document.querySelectorAll('.name-cell').forEach(c => { states[c.id] = c.dataset.state; });
        localStorage.setItem('att_remote_v1', JSON.stringify(states));
        alert('저장되었습니다.');
    }

    function load() {
        const states = JSON.parse(localStorage.getItem('att_remote_v1') || '{}');
        document.querySelectorAll('.name-cell').forEach(c => {
            if (states[c.id]) applyState(c, states[c.id]);
        });
    }

    function resetCheck() {
        if(confirm('모든 체크를 초기화하시겠습니까?')) {
            document.querySelectorAll('.name-cell').forEach(c => applyState(c, 0));
        }
    }

    // 앱 시작 시 구글 시트 데이터 로드
    window.onload = fetchSheetData;
</script>
